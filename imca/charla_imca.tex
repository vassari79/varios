% Draw a line in two styles: dashed from #1 to split point #2, solid from #2 to #3
\newcommand{\dashsolid}[3]{%
  \draw[thick, dashed] #1 -- #2;%
  \draw[thick]         #2 -- #3;%
}
% \angline{angle}{dashed-end dist}{solid-end dist}
% From origin: dashed from 0 to #2, solid from #2 to #3, all at angle #1 (degrees)
\newcommand{\angline}[3]{%
  \dashsolid{(0,0)}{({#2*cos(#1)},{#2*sin(#1)})}{({#3*cos(#1)},{#3*sin(#1)})}%
}
% \originline{angle}{dashed-length}{solid-length}
% From origin: dashed toward angle #1 for length #2, solid toward opposite direction for length #3
\newcommand{\originline}[3]{%
  \draw[thick, dashed] (0,0) -- ({#2*cos(#1)},{#2*sin(#1)});%
  \draw[thick]         (0,0) -- ({-#3*cos(#1)},{-#3*sin(#1)});%
}
